\section{通用盒子 (genbox)}

\begin{genbox}{内积空间的重要性质}
\begin{itemize}
    \item 毕达哥拉斯定理：若 \( \mathbf{u} \perp \mathbf{v} \)，则 \( \norm{\mathbf{u} + \mathbf{v}}^2 = \norm{\mathbf{u}}^2 + \norm{\mathbf{v}}^2 \)
    \item 平行四边形法则：\( \norm{\mathbf{u} + \mathbf{v}}^2 + \norm{\mathbf{u} - \mathbf{v}}^2 = 2(\norm{\mathbf{u}}^2 + \norm{\mathbf{v}}^2) \)
    \item 柯西-施瓦茨不等式：\( \abs{\langle \mathbf{u}, \mathbf{v} \rangle} \leq \norm{\mathbf{u}} \cdot \norm{\mathbf{v}} \)
\end{itemize}
\end{genbox}
