\section{引理环境 (lembox)}

\begin{lembox}[正交投影引理]{lem:projection}
设 \( W \) 是内积空间 \( V \) 的闭子空间，\( \mathbf{v} \in V \)，则存在唯一的 \( \mathbf{w} \in W \)，使得 \( \mathbf{v} - \mathbf{w} \perp W \)。此时 \( \mathbf{w} \) 称为 \( \mathbf{v} \) 在 \( W \) 上的正交投影。
\end{lembox}

\begin{lembox*}[线性无关引理]
若向量组 \( \{ \mathbf{v}_1, \dots, \mathbf{v}_k \} \) 线性无关，则其任意非空子集也线性无关。反之，若向量组包含线性相关的子集，则整个向量组线性相关。
\end{lembox*}

\begin{lembox}[基扩展引理]{lem:basis_extension}
设 \( \{ \mathbf{v}_1, \dots, \mathbf{v}_k \} \) 是 \( n \) 维线性空间 \( V \) 中的线性无关向量组，则存在 \( n - k \) 个向量 \( \{ \mathbf{v}_{k+1}, \dots, \mathbf{v}_n \} \)，使得 \( \{ \mathbf{v}_1, \dots, \mathbf{v}_n \} \) 构成 \( V \) 的一组基。
\end{lembox}
