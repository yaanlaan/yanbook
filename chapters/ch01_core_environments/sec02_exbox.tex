\section{示例环境 (exbox)}

\begin{exbox}[欧几里得空间]{ex:euclidean_space}
\( \mathbb{R}^n \) 是最常见的内积空间，其内积定义为：
\[
\langle \mathbf{u}, \mathbf{v} \rangle = \mathbf{u}^T \mathbf{v} = \sum_{i=1}^{n} u_i v_i
\]
\end{exbox}

\begin{exbox}[连续函数空间]{ex:continuous_space}
在区间 \([a, b]\) 上的连续函数集合 \( C[a, b] \) 构成内积空间，内积定义为：
\[
\langle f, g \rangle = \int_a^b f(x)g(x) dx
\]
\end{exbox}

\begin{exbox}[序列空间]{ex:sequence_space}
序列空间 \( \ell^2 \) 由所有平方可和的序列组成，内积定义为：
\[
\langle \{a_n\}, \{b_n\} \rangle = \sum_{n=1}^{\infty} a_n \overline{b_n}
\]
\end{exbox}
